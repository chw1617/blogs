# 时间复杂度

  • 定性地描述一个算法的运行时间
  • 是一个函数，用大O来表示，比如O(1)、O(n)、O(logn)
var i = 1
i += 1
//时间复杂度是O(1),上面代码自会执行一次

for(var i = 0;i<n;i++){
  console.log(n)
}
//时间复杂度是O(n),上代码会执行n次

var i = 1
i += 1
for(var i = 0;i<n;i++){
  console.log(n)
}
//时间复杂度是O(n),O(1) + O(n) = O(n) 只会统计增长次数更高的

var i = 1 
while(i<n){
  console.log(i)
  i * = 2
}
//时间复杂度为O(logN)

for(var i = 0;i<n;i++ ){
  for(var j=0;j<n;j++){
    console.log(i,j)
  }
}
//时间复杂度为O(n)*O(n)=O(n^2)

# 空间复杂度

  • 算法在运行的过程中占用存储空间的大小
  • 也是一个函数，用大O来表示，O(1),O(n),O(logN)
var i = 0
i += 1
// O(1),只声明了单个变量，占用空间为1

var list = []
for(var i = 0;i<n;i++){
  list.push(i)
}
// O(n)，不断往list增加n个值，那么空间大小就为n

var matrix = []
for(var i = 0;i<n;i++){
  matrix[i].push([])
  for(var j = 0;i<n;j++){
    matrix[i] .puhs(i,j)
  }
}
//O(n^2)，二维的矩阵

# 算法设计思想

  • 分而治之（分（两个独立的子问题，比如翻转二叉树），解（递归），合）
  • 动态规划（分（分成两个互相重叠的子问题，比如斐波那契数列），解（递归），合 ）
  • 贪心(局部最优解，从而追求最优解)
  • 回溯(渐进式寻找解决问题的方法，先从一个动作出发，不可以就回溯并选择另外一个动作出发) 全排列
